\documentclass{article}
\directlua{
Row = {}
Row.__index = Row
function Row:new(fn_next, start, stop, step)
    if type(fn_next) == "function" then
        step = step or 1
        if not stop then
            start, stop = 1, start
        end
    elseif type(fn_next) == "table" then
        local data = fn_next
        start, stop, step = 1, \string#data, 1
        fn_next = function(c, r)
            for i, v in ipairs(data[c]) do
                r[i] = v
            end
        end
    end
    local o = {
        fn_next = fn_next,
        start = start,
        stop = stop,
        step = step,
    }
    setmetatable(o, self)
    return o
end

function Row:next()
    local var = self.var
    if not var then
        var = self.start
    else
        var = var + self.step
    end
    if var <= self.stop then
        self.var = var
        local fn = self.fn_next
        fn(var, self)
        return true
    end
end
}

\directlua{
local data = {
    {"files.txt"     ,  4710},
    {"lib.lua"       ,   330},
    {"liteparse.txt",  6451},
    {"parse.lua"     , 50995},
    {"path.txt"      ,  2150},
    {"shapes.txt"    ,  4812},
    {"signature.txt" ,  1228},
}
row = Row:new(data)
}

\begin{document}
\begin{tabular}{lr}
\directlua{
local bs = string.char(92)
while row:next() do
tex.print(row[1].."&"..row[2]..bs..bs)
end
}
\end{tabular}
\end{document}
