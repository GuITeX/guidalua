local pw = {
    id = "PageWidth",
    info = function () return tex.pagewidth end,
    fmt = {
        point = function (pw, n)
            n = n or 2
            if n == 0 then
                return tostring(p)
            else
                assert(type(n), "number")
                assert(n>0)
                local p = string.char(37)
                local f1 = p..p.."0."..p.."df pt"
                local f2 = f1:format(n)
                return f2:format(pw/65536)
            end
        end
    },
}
lib_logger.register:add_to_log({pw, "point", 3}, 2)
}
